add_custom_command(OUTPUT ${CURRENT_DIR}/dcv2t.eps
  COMMAND ${CMAKE_COMMAND} -E copy_if_different ${CMAKE_CURRENT_SOURCE_DIR}/${CURRENT_DIR}/dcv2t.fig
    ${CMAKE_CURRENT_BINARY_DIR}/${CURRENT_DIR}/dcv2t.fig
  COMMAND ${CMAKE_COMMAND} -E copy_if_different ${CMAKE_CURRENT_SOURCE_DIR}/${CURRENT_DIR}/dcv2t_gaussian.eps
    ${CMAKE_CURRENT_BINARY_DIR}/${CURRENT_DIR}/dcv2t_gaussian.eps
  COMMAND ${CMAKE_COMMAND} -E copy_if_different ${CMAKE_CURRENT_SOURCE_DIR}/${CURRENT_DIR}/dcv2t_gromacs.eps
    ${CMAKE_CURRENT_BINARY_DIR}/${CURRENT_DIR}/dcv2t_gromacs.eps
  COMMAND ${CMAKE_COMMAND} -E copy_if_different ${CMAKE_CURRENT_SOURCE_DIR}/${CURRENT_DIR}/dcv2t_atom_types.eps
    ${CMAKE_CURRENT_BINARY_DIR}/${CURRENT_DIR}/dcv2t_atom_types.eps
  COMMAND ${FIGTEX2EPS} ${CMAKE_CURRENT_BINARY_DIR}/${CURRENT_DIR}/dcv2t.fig
    ${CMAKE_CURRENT_BINARY_DIR}/${CURRENT_DIR}/dcv2t.eps
  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/${CURRENT_DIR}
  DEPENDS ${CURRENT_DIR}/dcv2t.fig)
